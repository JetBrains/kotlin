@NonRestartableComposable @Composable
fun Example(items: Iterator<Int>) {
    // a group around while is needed here, but the function body group will suffice
    while (items.hasNext()) {
        val i = items.next()
        val j = i
        P(i)
        if (i == 0) {
            break
        }
        P(j)
    }
}
