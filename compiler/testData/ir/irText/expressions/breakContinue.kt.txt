fun test2() {
  OUTER@ while (true) { // BLOCK
    INNER@ while (true) { // BLOCK
      break@INNER
      break@OUTER
    }
    break@OUTER
  }
  OUTER@ while (true) { // BLOCK
    INNER@ while (true) { // BLOCK
      continue@INNER
      continue@OUTER
    }
    continue@OUTER
  }
}
