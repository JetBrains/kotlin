fun test1() {
  try { // BLOCK
    println()
  }
  catch (e: Throwable){ // BLOCK
    println()
  }
  finally { // BLOCK
    println()
  }
}

fun test2(): Int {
  return try { // BLOCK
    println()
    42
  }
  catch (e: Throwable){ // BLOCK
    println()
    24
  }
  finally { // BLOCK
    println()
    555 /*~> Unit */
  }
}
