@Composable
@ComposableInferredTarget(scheme = "[0[0]]")
fun Widget(content: Function2<Composer, Int, Unit>?, %composer: Composer?, %changed: Int) {
  %composer = %composer.startRestartGroup(<>)
  sourceInformation(%composer, "C(Widget)N(content):Test.kt")
  val %dirty = %changed
  if (%changed and 0b0110 == 0) {
    %dirty = %dirty or if (%composer.changedInstance(content)) 0b0100 else 0b0010
  }
  if (%composer.shouldExecute(%dirty and 0b0011 != 0b0010, %dirty and 0b0001)) {
    if (isTraceInProgress()) {
      traceEventStart(<>, %dirty, -1, <>)
    }
    if (content != null) {
      %composer.startReplaceGroup(<>)
      sourceInformation(%composer, "<conten...>")
      content as Function2<Composer, Int, Unit>(%composer, 0b1110 and %dirty)
      %composer.endReplaceGroup()
    } else {
      %composer.startReplaceGroup(<>)
      %composer.endReplaceGroup()
    }
    if (isTraceInProgress()) {
      traceEventEnd()
    }
  } else {
    %composer.skipToGroupEnd()
  }
  %composer.endRestartGroup()?.updateScope { %composer: Composer?, %force: Int ->
    Widget(content, %composer, updateChangedFlags(%changed or 0b0001))
  }
}
