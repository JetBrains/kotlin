public final class WhenIsKt {
    public static final fun foo(@org.jetbrains.annotations.NotNull bar: java.lang.Object) : java.lang.String {
        return switch (bar)  {
            it is java.lang.String -> {
                yield bar
            }

            it !is java.lang.String -> {
                yield "<error>"
            }

        }

    }
}
