enum class MyEnumClass {
  FirstEntry {
    fun a()

    override fun foo()
  },
  SecondEntry {
    override fun foo()

    override val i: kotlin.Int
      get()
  },
  ThirdEntry {
    override fun foo()

    override fun foo(i: kotlin.Int)
  };

  abstract fun foo()

  open fun foo(i: kotlin.Int)

  open val i: kotlin.Int
}
