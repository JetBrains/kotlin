interface I<T : Any?> {
  abstract fun input(t: T)

  abstract fun output(): T

}

fun <T : Any?> foo(i: I<(T & Any)>) {
  i.input(t = i.output())
}

fun <T : Any?> bar(i: I<out (T & Any)>): (T & Any) {
  return i.output()
}
